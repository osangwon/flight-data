    1. 시간대별 지연 분석

    def divide_hour(hour):
    if 6 <= hour < 12:
        return 'morning'
    if 12 <= hour < 18:
        return 'lunch'
    if 18 <= hour < 24:
        return 'dinner'
    return 'dawn'

flights['time_of_day'] = flights['hour'].apply(divide_hour)
result = flights.loc[:, ['hour', 'time_of_day']].head(3)
